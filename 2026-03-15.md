Go言語でルーティングの基礎を行いたく実装した。

``` go
package main

import (
	"fmt"
	"net/http"
)

/*
目的: サーバーを起動して、ブラウザに文字を表示する
やること
- :8080でサーバを起動する
- / にアクセスすると、ブラウザに「Hello, Pan-chan!」と表示する
*/

func homeHandler(w http.ResponseWriter, r *http.Request) {
	fmt.Fprint(w, "Hello, Pan-chan!")
}

func main() {
	mux := http.NewServeMux()
	mux.HandleFunc("/", homeHandler)

	fmt.Println("Server is running on http://localhost:8080")

	err := http.ListenAndServe(":8080", mux)
	if err != nil {
		panic(err)
	}
}
```
