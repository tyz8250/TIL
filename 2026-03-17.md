# Go TIL - html/template の基本

Goでは`html/template`を使用することで、HTMLファイルをテンプレートとして表示できる。

## なぜrender()なのか　、まだ腑におちていない




``` go
// renderの流れを考えてみる

// appがtemplateDirを持つ
// render()がHTMLを読む
// Execute()でwに書き込む
// home()がrender()を呼ぶ

package main

import (
	"fmt"
	"net/http"
	"path/filepath"
	"text/template"
)

// テンプレートディレクトリを管理する構造体
type application struct {
	templateDir string
}

func (app *application) render(w http.ResponseWriter, file string, data any) error {
	// テンプレートファイルのパスを構築
	filePath := filepath.Join(app.templateDir, file)

	// HTMLファイルをテンプレートして読み込む
	tmpl, err := template.ParseFiles(filePath)
	if err != nil {
		return err
	}

	// テンプレートを実行してレスポンスに書き込む
	return tmpl.Execute(w, data)
}

func (app *application) home(w http.ResponseWriter, r *http.Request) {
	err := app.render(w, "index.html", nil)
	if err != nil {
		http.Error(w, "template error", http.StatusInternalServerError)
		return
	}
}

func main() {
	app := &application{
		templateDir: "templates",
	}

	mux := http.NewServeMux()
	mux.HandleFunc("/", app.home)

	fmt.Println("server started at :8080")
	if err := http.ListenAndServe(":8080", mux); err != nil {
		panic(err)
	}
}
```
