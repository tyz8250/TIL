ルーティングとCookieの練習

```go
package main

import (
	"fmt"
	"net/http"
)

// アクセス回数カウンターAPI

var count int

func homeHandler(w http.ResponseWriter, r *http.Request) {
	fmt.Fprintln(w, "アクセスカウンターAPIへようこそ！")
}

// /countにアクセスしたら増やす
func countHandler(w http.ResponseWriter, r *http.Request) {
	count++
	fmt.Fprintf(w, "アクセス回数: %d", count)
}

// リセット機能
func resetHandler(w http.ResponseWriter, r *http.Request) {
	count = 0
	fmt.Fprintln(w, "カウンターをリセットしました")
}
go
// ログイン機能（みせかけ）
func loginHandler(w http.ResponseWriter, r *http.Request) {
	name := r.URL.Query().Get("name")

	cookie := &http.Cookie{
		Name:  "username",
		Value: name,
	}

	http.SetCookie(w, cookie)
	fmt.Fprint(w, "ログイン成功！", name)
}

// meHandlerはログインしているユーザーの名前を返す
func meHandler(w http.ResponseWriter, r *http.Request) {
	cookie, err := r.Cookie("username")
	if err != nil {
		fmt.Fprintln(w, "ログインしていません")
		return
	}

	fmt.Fprintf(w, "あなたは %s さんです", cookie.Value)
}

func main() {
	mux := http.NewServeMux()
	mux.HandleFunc("/count", countHandler)
	mux.HandleFunc("/reset", resetHandler)
	mux.HandleFunc("/login", loginHandler)
	mux.HandleFunc("/me", meHandler)

	fmt.Println("サーバを起動")
	err := http.ListenAndServe(":8080", mux)
	if err != nil {
		panic(err)
	}
}
```
